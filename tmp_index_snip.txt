          banner.hidden = false;
          banner.innerHTML = `<i class="fa-solid fa-user-group"></i><span>You are viewing @${referral}'s invite link. Create an account to start building your own Zo2y profile.</span>`;
        } else {
          banner.hidden = true;
        }
      }
      document.querySelectorAll('[data-auth-entry]').forEach((link) => {
        const href = link.getAttribute('href');
        if (!href) return;
        link.setAttribute('href', referralUtils.appendReferralToHref(href, { referral, next: 'index.html' }));
      });
